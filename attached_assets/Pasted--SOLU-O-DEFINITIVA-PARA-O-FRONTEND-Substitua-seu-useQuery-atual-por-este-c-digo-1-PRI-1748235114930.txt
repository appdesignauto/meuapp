        console.error('âŒ [QUERY] Erro HTTP:', response.status, errorText);

        console.log('ğŸ” [QUERY] Propriedades disponÃ­veis:', Object.keys(data));

      return data;
      
    } catch (error) {
      console.error('âŒ [QUERY] Erro na requisiÃ§Ã£o:', error);
      throw error;
    }
