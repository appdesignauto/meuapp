        class DesignAutoDarkMode {
            constructor() {
                this.currentTheme = 'light';
                this.init();
            }

            init() {
                // Detectar tema salvo ou prefer√™ncia do sistema
                const savedTheme = localStorage.getItem('designauto-theme');
                const systemPrefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
                
                this.currentTheme = savedTheme || (systemPrefersDark ? 'dark' : 'light');
                this.applyTheme(this.currentTheme);
                
                // Escutar mudan√ßas do sistema
                window.matchMedia('(prefers-color-scheme: dark)')
                    .addEventListener('change', (e) => {
                        if (!localStorage.getItem('designauto-theme')) {
                            this.currentTheme = e.matches ? 'dark' : 'light';
                            this.applyTheme(this.currentTheme);
                        }
                    });

                // Atalho de teclado
                document.addEventListener('keydown', (e) => {
                    if ((e.ctrlKey || e.metaKey) && e.shiftKey && e.key === 'D') {
                        e.preventDefault();
                        this.toggle();
                    }
                });
            }

            applyTheme(theme) {
                document.documentElement.setAttribute('data-theme', theme);
                localStorage.setItem('designauto-theme', theme);
                this.currentTheme = theme;
                
                // Feedback visual
                this.showNotification(
                    theme === 'dark' ? 'üåô Modo escuro ativado' : '‚òÄÔ∏è Modo claro ativado'
                );
            }

            toggle() {
                const newTheme = this.currentTheme === 'light' ? 'dark' : 'light';
                this.applyTheme(newTheme);
                return newTheme;
            }
